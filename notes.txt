# Create dependencies with venv
pip freeze > requirements.txt

# Delete __pycache__ directories
find . -type d -name "__pycache__" -exec rm -rf {} +

#install dep without versions
pip install $(grep -vE '^\s*(#|$)' requirements.txt \
            | sed -E 's/(==|>=|<=|~=|!=).*$//' \
            | sort -u)
